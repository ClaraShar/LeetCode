class Solution {
public:
    vector<int> grayCode(int n) {
        vector<int> code(pow(2,n));
        code[0]=0;
        code[1]=1;
        if(n==1)
            return code;
        else
        {
            int step=2;
            while(step<=n)
            {
                int num=pow(2,step-1);
                for(int i=0,j=pow(2,step)-1;i<num;i++,j--)
                    code[j]=code[i]+num;
                step++;
            }
        }
        return code;
    }
};
